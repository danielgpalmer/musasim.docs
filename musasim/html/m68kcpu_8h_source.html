01164 /* Indexed addressing modes are encoded as follows:
01165  *
01166  * Base instruction format:
01167  * F E D C B A 9 8 7 6 | 5 4 3 | 2 1 0
01168  * x x x x x x x x x x | 1 1 0 | BASE REGISTER      (An)
01169  *
01170  * Base instruction format for destination EA in move instructions:
01171  * F E D C | B A 9    | 8 7 6 | 5 4 3 2 1 0
01172  * x x x x | BASE REG | 1 1 0 | X X X X X X       (An)
01173  *
01174  * Brief extension format:
01175  *  F  |  E D C   |  B  |  A 9  | 8 | 7 6 5 4 3 2 1 0
01176  * D/A | REGISTER | W/L | SCALE | 0 |  DISPLACEMENT
01177  *
01178  * Full extension format:
01179  *  F     E D C      B     A 9    8   7    6    5 4       3   2 1 0
01180  * D/A | REGISTER | W/L | SCALE | 1 | BS | IS | BD SIZE | 0 | I/IS
01181  * BASE DISPLACEMENT (0, 16, 32 bit)                (bd)
01182  * OUTER DISPLACEMENT (0, 16, 32 bit)               (od)
01183  *
01184  * D/A:     0 = Dn, 1 = An                          (Xn)
01185  * W/L:     0 = W (sign extend), 1 = L              (.SIZE)
01186  * SCALE:   00=1, 01=2, 10=4, 11=8                  (*SCALE)
01187  * BS:      0=add base reg, 1=suppress base reg     (An suppressed)
01188  * IS:      0=add index, 1=suppress index           (Xn suppressed)
01189  * BD SIZE: 00=reserved, 01=NULL, 10=Word, 11=Long  (size of bd)
01190  *
01191  * IS I/IS Operation
01192  * 0  000  No Memory Indirect
01193  * 0  001  indir prex with null outer
01194  * 0  010  indir prex with word outer
01195  * 0  011  indir prex with long outer
01196  * 0  100  reserved
01197  * 0  101  indir postx with null outer
01198  * 0  110  indir postx with word outer
01199  * 0  111  indir postx with long outer
01200  * 1  000  no memory indirect
01201  * 1  001  mem indir with null outer
01202  * 1  010  mem indir with word outer
01203  * 1  011  mem indir with long outer
01204  * 1  100-111  reserved
01205  */
01206 INLINE uint m68ki_get_ea_ix(uint An)
01207 {
01208         /* An = base register */
01209         uint extension = m68ki_read_imm_16();
01210         uint Xn = 0;                        /* Index register */
01211         uint bd = 0;                        /* Base Displacement */
01212         uint od = 0;                        /* Outer Displacement */
01213 
01214         if(CPU_TYPE_IS_010_LESS(CPU_TYPE))
01215         {
01216                 /* Calculate index */
01217                 Xn = REG_DA[extension>>12];     /* Xn */
01218                 if(!BIT_B(extension))           /* W/L */
01219                         Xn = MAKE_INT_16(Xn);
01220 
01221                 /* Add base register and displacement and return */
01222                 return An + Xn + MAKE_INT_8(extension);
01223         }
01224 
01225         /* Brief extension format */
01226         if(!BIT_8(extension))
01227         {
01228                 /* Calculate index */
01229                 Xn = REG_DA[extension>>12];     /* Xn */
01230                 if(!BIT_B(extension))           /* W/L */
01231                         Xn = MAKE_INT_16(Xn);
01232                 /* Add scale if proper CPU type */
01233                 if(CPU_TYPE_IS_EC020_PLUS(CPU_TYPE))
01234                         Xn <<= (extension>>9) & 3;  /* SCALE */
01235 
01236                 /* Add base register and displacement and return */
01237                 return An + Xn + MAKE_INT_8(extension);
01238         }
01239 
01240         /* Full extension format */
01241 
01242         USE_CYCLES(m68ki_ea_idx_cycle_table[extension&0x3f]);
01243 
01244         /* Check if base register is present */
01245         if(BIT_7(extension))                /* BS */
01246                 An = 0;                         /* An */
01247 
01248         /* Check if index is present */
01249         if(!BIT_6(extension))               /* IS */
01250         {
01251                 Xn = REG_DA[extension>>12];     /* Xn */
01252                 if(!BIT_B(extension))           /* W/L */
01253                         Xn = MAKE_INT_16(Xn);
01254                 Xn <<= (extension>>9) & 3;      /* SCALE */
01255         }
01256 
01257         /* Check if base displacement is present */
01258         if(BIT_5(extension))                /* BD SIZE */
01259                 bd = BIT_4(extension) ? m68ki_read_imm_32() : MAKE_INT_16(m68ki_read_imm_16());
01260 
01261         /* If no indirect action, we are done */
01262         if(!(extension&7))                  /* No Memory Indirect */
01263                 return An + bd + Xn;
01264 
01265         /* Check if outer displacement is present */
01266         if(BIT_1(extension))                /* I/IS:  od */
01267                 od = BIT_0(extension) ? m68ki_read_imm_32() : MAKE_INT_16(m68ki_read_imm_16());
01268 
01269         /* Postindex */
01270         if(BIT_2(extension))                /* I/IS:  0 = preindex, 1 = postindex */
01271                 return m68ki_read_32(An + bd) + Xn + od;
01272 
01273         /* Preindex */
01274         return m68ki_read_32(An + bd + Xn) + od;
01275 }
