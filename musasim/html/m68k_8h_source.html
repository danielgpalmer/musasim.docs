00127 /* read/write functions called by the CPU to access memory.
00128  * while values used are 32 bits, only the appropriate number
00129  * of bits are relevant (i.e. in write_memory_8, only the lower 8 bits
00130  * of value should be written to memory).
00131  *
00132  * NOTE: I have separated the immediate and PC-relative memory fetches
00133  *       from the other memory fetches because some systems require
00134  *       differentiation between PROGRAM and DATA fetches (usually
00135  *       for security setups such as encryption).
00136  *       This separation can either be achieved by setting
00137  *       M68K_SEPARATE_READS in m68kconf.h and defining
00138  *       the read functions, or by setting M68K_EMULATE_FC and
00139  *       making a function code callback function.
00140  *       Using the callback offers better emulation coverage
00141  *       because you can also monitor whether the CPU is in SYSTEM or
00142  *       USER mode, but it is also slower.
00143  */

00175 /* These functions allow you to set callbacks to the host when specific events
00176  * occur.  Note that you must enable the corresponding value in m68kconf.h
00177  * in order for these to do anything useful.
00178  * Note: I have defined default callbacks which are used if you have enabled
00179  * the corresponding #define in m68kconf.h but either haven't assigned a

00183 /* Set the callback for an interrupt acknowledge.
00184  * You must enable M68K_EMULATE_INT_ACK in m68kconf.h.
00185  * The CPU will call the callback with the interrupt level being acknowledged.
00186  * The host program must return either a vector from 0x02-0xff, or one of the
00187  * special interrupt acknowledge values specified earlier in this header.
00188  * If this is not implemented, the CPU will always assume an autovectored
00189  * interrupt, and will automatically clear the interrupt request when it
00190  * services the interrupt.
00191  * Default behavior: return M68K_INT_ACK_AUTOVECTOR.
00192  */
00193 void m68k_set_int_ack_callback(int  (*callback)(int int_level));
00194 
00195 
00196 /* Set the callback for a breakpoint acknowledge (68010+).
00197  * You must enable M68K_EMULATE_BKPT_ACK in m68kconf.h.
00198  * The CPU will call the callback with whatever was in the data field of the
00199  * BKPT instruction for 68020+, or 0 for 68010.
00200  * Default behavior: do nothing.
00201  */
00202 void m68k_set_bkpt_ack_callback(void (*callback)(unsigned int data));
